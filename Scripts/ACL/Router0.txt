!
! =====================================================
! Router0 — Core Router Configuration
! ACL Lab: Block Students from reaching the Server
! Routing: OSPFv2 Area 0
! =====================================================
!
! --- BASIC SETUP ---
!
enable
configure terminal
!
hostname Router0
no ip domain-lookup
!
banner motd #
*** Authorized Access Only – Router0 (Core) ***
#
!
!
!
! --- INTERFACE CONFIGURATION ---
!
interface GigabitEthernet0/0
 description Link_To_Server_30.0.0.0/8
 ip address 30.0.0.1 255.0.0.0
 ip access-group BlockStudents out
 ! ACL applied OUTBOUND — blocks Student traffic before it hits the Server
 no shutdown
!
interface GigabitEthernet0/1
 description Link_To_Router2_Teachers_192.168.1.4/30
 ip address 192.168.1.5 255.255.255.252
 no shutdown
!
interface GigabitEthernet0/2
 description Link_To_Router1_Students_192.168.1.0/30
 ip address 192.168.1.2 255.255.255.252
 no shutdown
!
! --- ROUTING: OSPFv2 Area 0 ---
!
router ospf 1
 network 30.0.0.0 0.255.255.255 area 0
 network 192.168.1.0 0.0.0.3 area 0
 network 192.168.1.4 0.0.0.3 area 0
!
! --- TASK: STANDARD NAMED ACL (BlockStudents) ---
!
ip access-list standard BlockStudents
 deny 10.0.0.0 0.255.255.255
 ! Denies ALL traffic from Students network (10.0.0.0/8)
 permit any
 ! Permits all other traffic (Teachers, management, etc.)
!
end
write memory
