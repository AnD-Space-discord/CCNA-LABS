enable
configure terminal
hostname R1
no ip domain-lookup

interface GigabitEthernet0/0
description "LAN Network"
ip address 192.168.10.1 255.255.255.0
no shutdown
exit
interface GigabitEthernet0/1
description "Server Network"
ip address 10.0.0.1 255.255.255.0
no shutdown
exit

ip access-list extended BLOCK_WEB_PING
deny tcp 192.168.10.0 0.0.0.255 host 10.0.0.10 eq 80
deny icmp 192.168.10.0 0.0.0.255 host 10.0.0.10 echo
permit ip any any
exit

interface GigabitEthernet0/0
ip access-group BLOCK_WEB_PING in
exit
